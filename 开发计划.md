```js
(function() {
  'use strict';

  // === DOM 引用 ===
  const appEl = document.getElementById('app');
  const inputEl = document.getElementById('option-input');
  const tagsEl = document.getElementById('tags-container');
  const btnStart = document.getElementById('btn-start');

  // === 标签颜色列表（循环使用） ===
  const TAG_COLORS = ['yellow', 'pink', 'mint', 'peach'];

  // === 渲染标签 ===
  function renderTags(options) {
    if (!options || options.length === 0) {
      tagsEl.innerHTML = '<span class="tags-empty">输入选项后将在此显示标签</span>';
      btnStart.classList.add('btn-disabled');
      return;
    }
    tagsEl.innerHTML = options.map((opt, i) =>
      `<span class="tag ${TAG_COLORS[i % TAG_COLORS.length]}">${escapeHTML(opt)} <span class="tag-close" data-index="${i}">✕</span></span>`
    ).join('');
    // 控制按钮状态
    if (options.length >= CONSTANTS.OPTION_MIN) {
      btnStart.classList.remove('btn-disabled');
    } else {
      btnStart.classList.add('btn-disabled');
    }
  }

  // === 输入事件 ===
  inputEl.addEventListener('input', function() {
    const options = parseOptions(this.value);
    renderTags(options);
    Store.quiz.options = options;
  });

  // === 点击标签关闭按钮删除选项 ===
  tagsEl.addEventListener('click', function(e) {
    if (e.target.classList.contains('tag-close')) {
      const idx = parseInt(e.target.dataset.index);
      Store.quiz.options.splice(idx, 1);
      inputEl.value = Store.quiz.options.join(', ');
      renderTags(Store.quiz.options);
    }
  });

  // === 开始决策按钮 ===
  btnStart.addEventListener('click', function() {
    if (this.classList.contains('btn-disabled')) return;
    if (Store.quiz.options.length < CONSTANTS.OPTION_MIN) {
      alert('请至少输入 2 个选项');
      return;
    }
    if (Store.quiz.options.length > CONSTANTS.OPTION_MAX) {
      alert(`最多支持 ${CONSTANTS.OPTION_MAX} 个选项`);
      return;
    }
    // 切换到问答页（Step 2 实现，当前先 alert 占位）
    Store.currentPage = 'quiz';
    alert('即将进入问答页（Step 2 实现）');
  });
})();
```

#### 2.1 创建 `js/router.js`

实现简易页面切换（不使用 hash 路由以保持简单，用函数直接切换）：

```js
const Router = {
  /**
   * 切换到指定页面
   * @param {string} page - 'home' | 'quiz' | 'result' | 'history' | 'admin'
   * @param {object} data - 传递给目标页面的数据（可选）
   */
  navigate(page, data) {
    Store.currentPage = page;
    // 隐藏所有页面
    document.querySelectorAll('.page-section').forEach(el => el.style.display = 'none');
    // 显示目标页面
    const target = document.getElementById('page-' + page);
    if (target) target.style.display = '';
    // 触发页面初始化
    if (typeof window['init' + page.charAt(0).toUpperCase() + page.slice(1)] === 'function') {
      window['init' + page.charAt(0).toUpperCase() + page.slice(1)](data);
    }
  }
};
```

#### 2.2 创建 `js/services/questionService.js`

```js
const QuestionService = {
  /**
   * 从 Store.questions 中选取本轮题目：
   * 1. 筛选 enabled === true
   * 2. 按 weight 降序排列
   * 3. 取前 MAX_QUESTIONS_PER_QUIZ 题
   * @returns {object[]}
   */
  selectQuestions() {
    const enabled = Store.questions.filter(q => q.enabled);
    enabled.sort((a, b) => b.weight - a.weight);
    return enabled.slice(0, CONSTANTS.MAX_QUESTIONS_PER_QUIZ);
  },

  /**
   * 从 questions.json 加载问题到 Store
   * 使用 fetch，如果失败则用内联默认数据
   */
  async loadQuestions() {
    try {
      const resp = await fetch('data/questions.json');
      const data = await resp.json();
      Store.questions = data.questions || [];
    } catch (e) {
      console.warn('无法加载 questions.json，使用默认问题');
      Store.questions = getDefaultQuestions();  // 硬编码的 5 道题
```

```js
function renderBarChart(canvasId, options, finalScores, winner) {
  const ctx = document.getElementById(canvasId);

        data: scores,
        backgroundColor: bgColors,
        borderColor: '#1A1A1A',
        borderWidth: 3,
        borderRadius: 8,
        borderSkipped: false,
        barPercentage: 0.55,
        categoryPercentage: 0.7

    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      plugins: {
        legend: { display: false },
        tooltip: {
          backgroundColor: '#FFFBF5',
          titleColor: '#1A1A1A',
          bodyColor: '#3A3A3A',
          borderColor: '#1A1A1A',
          borderWidth: 3,
          padding: 14,
          displayColors: false,
          callbacks: { label: ctx => ' 得分：' + ctx.raw + ' 分' }
        }
      },
      scales: {
        x: {
          ticks: { color: '#3A3A3A', font: { size: 14, weight: 'bold' } },
          grid: { display: false },
          border: { color: '#1A1A1A', width: 2.5 }
        },
        y: {

          ticks: {
            color: '#7A7A7A',
            font: { size: 12, weight: 'bold' },
            callback: v => v + ' 分'
          },
          grid: { color: 'rgba(0,0,0,0.05)' },
          border: { color: '#1A1A1A', width: 2.5 }
        }
      },
      animation: { duration: 800, easing: 'easeOutBounce' }
    }
  });
}
```

```js
const ExportService = {
  /**
   * 导出为 JSON 文件并触发下载
   * 文件名格式：决策历史_2026-06-10.json
   */
  exportJSON(records) {
    const blob = new Blob([JSON.stringify(records, null, 2)], { type: 'application/json' });
    this._download(blob, '决策历史_' + this._dateStr() + '.json');
  },

  /**
   * 导出为 CSV 文件并触发下载
   * 列：时间, 选项数, 选项列表, 推荐结果, 推荐得分, 推荐理由
   */
  exportCSV(records) {
    const header = '时间,选项数,选项列表,推荐结果,推荐得分,推荐理由\n';
    const rows = records.map(r => {
      return [
        r.timestamp,
        r.options.length,
        '"' + r.options.join('、') + '"',
        r.recommendation,
        r.finalScores[r.recommendation],
        '"' + (r.reason || '').replace(/"/g, '""') + '"'
      ].join(',');
    }).join('\n');
    // BOM for Excel UTF-8 compatibility
    const blob = new Blob(['﻿' + header + rows], { type: 'text/csv;charset=utf-8' });
    this._download(blob, '决策历史_' + this._dateStr() + '.csv');
  },

  /**
   * 触发浏览器下载
   */
  _download(blob, filename) {
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  },

  /**
   * 今日日期字符串：2026-06-10
   */
  _dateStr() {
    return new Date().toISOString().slice(0, 10);
  }
};
```

## 附录 C：关键数据流

### C.1 一次完整决策的数据流

```
用户输入 "火锅, 烧烤, 日料"
  → parser.js 解析 → ["火锅", "烧烤", "日料"]
  → 存入 Store.quiz.options
  → 点击「开始决策」
  → QuestionService.selectQuestions() 选题（最多5道，按权重降序）
  → 存入 Store.quiz.questions
  → 问答页逐题展示 + 用户拖动滑块打分
  → Store.quiz.scores = { q001: {火锅:2, 烧烤:3, 日料:5}, ... }
  → 最后一题 → 点击「查看结果」
  → ScoreEngine.run(options, questions, scores)
    → calculateFinalScores → {火锅:19, 烧烤:21, 日料:22}
    → selectWinner → {winner: "日料", runnerUp: "烧烤", gap:1, ...}
  → ReasonGenerator.generate(...) → "在3个选项中..."
  → 结果页渲染：
    - 推荐选项 pill
    - 推荐理由 text
    - 柱状图（Chart.js）
    - 得分明细表
  → 用户点击「保存到历史」
  → HistoryService.save(record) → localStorage
```

### C.2 数据存储

```
localStorage:
  ┌─────────────────────────┐
  │ ct_questions            │  ← 问题库（管理员修改后存这里，优先于 JSON 文件）
  │ ct_history              │  ← 历史记录数组，最多 10 条
  │ ct_config               │  ← 用户配置（密码、AI设置等）
  │ ct_questions_backup     │  ← 问题库备份
  └─────────────────────────┘

data/ JSON 文件（静态）:
  ┌─────────────────────────┐
  │ questions.json          │  ← 首次加载时的默认问题库
  │ history.json            │  ← 初始空模板
  │ config.json             │  ← 默认配置
  └─────────────────────────┘
```

**优先级**：localStorage > JSON 文件。首次使用时从 JSON 文件加载，之后读写 localStorage，JSON 文件作为只读默认值。
